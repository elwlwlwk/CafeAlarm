function dynamicParamLink(reqURL, event){
	var noticeHidden = $('notice_hidden');

	
	if(event) {	
		if(event.ctrlKey || event.metaKey) {
			return;
		}
	}
	 
	if(noticeHidden == null) {
		document.location.href = reqURL;
		return false;
	} 	
	
	var isNoticeClosed = $('notice_hidden').checked;
	 
	if(isNoticeClosed) {
		reqURL = reqURL + "&noticeHidden=true";
		document.location.href = reqURL ;
		return false;
	}
	
	document.location.href = reqURL;
	return false;
}
